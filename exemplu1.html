Problema pentru un numar:

Transformarea unui numar citit de la tastatura in baza 10 in baza b:

#include 

using namespace std;

int main()
{
    long long n, b, x = 0, c, p = 1;
    
    cin >> n >> b; // Se citesc de la tastatura numarul n si baza b 
    
    while (n != 0) 
    {
        c = n % b;
        x = x + p * c; // Construim numarul in baza b
        p = p * 10;
        n = n / b;
    }
    
    cout << x;
    
    return 0;
}
