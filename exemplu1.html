Problema pentru un numar:

Se citeste un numar in baza 10 n si o baza b de la tastatura. Sa se afiseze suma cifrelor numarului n in baza b.

#include 

using namespace std;

int main()
{
    long long n, b, x = 0, c, p = 1, s = 0;
    
    cin >> n >> b; // Se citesc de la tastatura numarul n si baza b 
    
    while (n != 0) 
    {
        c = n % b;
        x = x + p * c; // Construim numarul in baza b
        p = p * 10;
        n = n / b;
    }
    
    while (x != 0)
    {
        s = s + x % 10;
        x = x / 10;
    }

    cout << s;
    
    return 0;
}
