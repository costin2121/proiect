Problema pentru n numere:

Transformarea unui numar citit de la tastatura in baza 10 in baza b:

#include 

using namespace std;

int main()
{
    long long n, nr, b, x = 0, c, p = 1;
    
    cin >> n >> b; // Se citesc de la tastatura numarul de numere si baza
    
    for (int i = 0; i < n; i++)
    {
        cin >> nr;

        while (nr != 0) 
        {
            c = nr % b;
            x = x + p * c; // Construim numarul in baza b
            p = p * 10;
            nr = nr / b;
        }
        
        cout << x << " ";

        x = 0;
        p = 1;
    }

    cout << x;
    return 0;
}
