Problema pentru n numere:

Se citeste un sir de n numere naturale de la tastatura, si o baza b. Sa se afiseze doar numerele din sir a caror reprezentare in baza b contine doar cifre pare.

#include 

using namespace std;

int main()
{
    long long n, nr, b, x = 0, c, p = 1, ok = 1, k = 0;
    
    cin >> n >> b; // Se citesc de la tastatura numarul de numere si baza
    
    for (int i = 0; i < n; i++)
    {
        cin >> nr;

        while (nr != 0 && ok == 1) 
        {
            c = nr % b;
            x = x + p * c; // Construim numarul in baza b
            p = p * 10;
            nr = nr / b;
            if (c % 2 != 0) ok = 0; // Daca numarul nu este par, se opreste bucla
        }
        
        if (ok == 1) {
            cout << x << " "; // Daca numarul este par, se afiseaza
            k++;
        }

        x = 0;
        p = 1;
        ok = 1;
    }

    if (k == 0) cout << "Nu exista numere cu cifre pare.";
    return 0;
}
